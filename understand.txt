3. Books.js - Admin Book Management
javascript
const Books = () => {
  const [books, setBooks] = useState([]);
  
  // 1. Load books on page open
  useEffect(() => {
    axios.get("http://localhost:5000/api/books")
      .then(res => setBooks(res.data));
  }, []);
  
  // 2. Delete book function
  const handleDelete = async (id) => {
    if (confirm("Delete this book?")) {
      await axios.delete(`http://localhost:5000/api/books/${id}`);
      setBooks(books.filter(book => book.BookID !== id)); // Remove from screen
    }
  };
  
  return (
    <div>
      <h1>Book Management</h1>
      <table>
        {books.map(book => (
          <tr key={book.BookID}>
            <td>{book.Title}</td>
            <td>{book.AuthorName}</td>
            <td>
              <button onClick={() => handleDelete(book.BookID)}>Delete</button>
              <Link to={`/update/${book.BookID}`}>Update</Link>
            </td>
          </tr>
        ))}
      </table>
    </div>
  );
};
Translation: Shows table of all books â†’ Admin can delete or update each one

const ProtectedRoute = ({ children }) => {
  const token = localStorage.getItem('token');
  
  if (!token) {
    return <Navigate to="/login" />; // Go to login if no token
  }
  
  return children; // Show the page if token exists
};

ðŸ”§ If Professor Asks Technical Questions:
Q: "How do you handle images?"
A: "Multer saves uploaded files to backend/images/, converts to base64 for database storage, and serves them as data URLs to the frontend."

Q: "How does search work without reloading?"
A: "React filters the existing book array locally as the user types, providing instant results without API calls."

Q: "Why JWT instead of sessions?"
A: "JWT tokens are stateless and work well with React's single-page architecture. The token contains user info and is easily verified on each request."

Q: "How do you prevent unauthorized access?"
A: "Protected routes check for valid JWT tokens, and backend endpoints verify tokens before processing sensitive operations like delete/update."
